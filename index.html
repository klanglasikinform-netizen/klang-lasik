// Node.js — ส่ง Push Message ด้วย LINE Messaging API
const axios = require('axios');

async function sendLineMessage(userId, message) {
  const token = 'YOUR_CHANNEL_ACCESS_TOKEN';
  try {
    const response = await axios.post(
      'https://api.line.me/v2/bot/message/push',
      { to: userId, messages: [{ type: 'text', text: message }] },
      { headers: { 'Content-Type': 'application/json', 'Authorization': `Bearer ${token}` } }
    );
    console.log('ส่งสำเร็จ:', response.data);
  } catch (err) {
    console.error('เกิดข้อผิดพลาด:', err.response?.data);
  }
}

# Python — ส่ง Push Message ด้วย LINE Messaging API
import requests, json

def send_line_message(user_id, message):
    token = "YOUR_CHANNEL_ACCESS_TOKEN"
    headers = { "Content-Type": "application/json", "Authorization": f"Bearer {token}" }
    payload = { "to": user_id, "messages": [{ "type": "text", "text": message }] }
    res = requests.post("https://api.line.me/v2/bot/message/push", headers=headers, data=json.dumps(payload))
    print("ส่งสำเร็จ!" if res.status_code == 200 else f"Error: {res.status_code} {res.json()}")

// Google Apps Script — ส่ง LINE Message อัตโนมัติ
// ตั้ง Trigger: Time-driven → Day timer → 08:00–09:00

const CHANNEL_TOKEN = 'YOUR_CHANNEL_ACCESS_TOKEN';
const SHEET_ID = 'YOUR_GOOGLE_SHEET_ID';
const REMIND_DAYS = 1;

function sendDailyReminders() {
  const target = new Date();
  target.setDate(target.getDate() + REMIND_DAYS);
  const targetStr = Utilities.formatDate(target, "Asia/Bangkok", "yyyy-MM-dd");
  const data = SpreadsheetApp.openById(SHEET_ID).getActiveSheet().getDataRange().getValues();
  // Headers: id, name, userId, doctor, type, date, time, status
  for (let i = 1; i < data.length; i++) {
    const [id, name, userId, doctor, type, date, time, status] = data[i];
    if (date === targetStr && status === 'confirmed') {
      sendPushMessage(userId, `🏥 เตือนความจำ!\nคุณ ${name} มีนัดตรวจตา LASIK พรุ่งนี้\n📅 ${date} เวลา ${time} น.\n👨‍⚕️ ${doctor} | ${type}\n🔢 ${id}`);
    }
  }
}

function sendPushMessage(userId, message) {
  const res = UrlFetchApp.fetch('https://api.line.me/v2/bot/message/push', {
    method: 'post', contentType: 'application/json',
    headers: { Authorization: `Bearer ${CHANNEL_TOKEN}` },
    payload: JSON.stringify({ to: userId, messages: [{ type: 'text', text: message }] }),
    muteHttpExceptions: true
  });
  Logger.log('Status: ' + res.getResponseCode());
}

🤔 Line Notify กับ Messaging API ต่างกันอย่างไร? +
Line Notify ส่งผ่านบัญชีสาธารณะ "Line Notify" และ ปิดบริการ 31 มีนาคม 2568

Messaging API ส่งผ่าน Official Account ของคลินิกเอง มีฟีเจอร์ครบกว่า และยังใช้งานได้ต่อเนื่อง — แนะนำให้ใช้ตัวนี้
💰 Line Messaging API เสียเงินไหม? +
ฟรีแพลน: ส่งได้ 200 Push Messages/เดือน ฟรี เพียงพอสำหรับคลินิกที่มีนัดไม่เกิน 200 คน/เดือน

Paid plan: ประมาณ 5 USD (~175 บาท) สำหรับ 1,000 ข้อความเพิ่ม
📱 ผู้ป่วยต้องทำอะไรเพิ่มเติมไหม? +
ผู้ป่วยต้อง "เพิ่มเพื่อน" (Add Friend) กับ Official Account ของคลินิกก่อน จึงสามารถรับข้อความได้

วิธีแก้: ติดป้าย QR Code ของ OA ในคลินิก หรือส่ง Link เพิ่มเพื่อนในอีเมลยืนยันนัด
🔒 ข้อมูลผู้ป่วยปลอดภัยไหมเมื่อส่งผ่าน LINE? +
LINE ใช้การเข้ารหัส SSL/TLS ปลอดภัยในระดับที่ยอมรับได้ แนะนำ: ไม่ส่งข้อมูลอ่อนไหวผ่าน LINE เช่น ผลตรวจโดยละเอียด และตรวจสอบว่าผู้ป่วยยินยอมรับข้อมูลทาง LINE ตาม PDPA
⚙️ ถ้าไม่มีโปรแกรมเมอร์ จะตั้งค่าเองได้ไหม? +
ได้ — วิธีง่ายที่สุด:
1. ใช้ Google Apps Script (ฟรี) — ดูโค้ดตัวอย่างในแท็บ "ทดสอบ"
2. ใช้ Make หรือ Zapier — เชื่อมต่อแบบ Drag & Drop ไม่ต้องเขียนโค้ด
3. Google Sheet + Apps Script Trigger ทุกเช้าส่งได้เลย
🚫 เกิดอะไรขึ้นถ้า Token หมดอายุหรือ Revoke? +
Channel Access Token แบบ Long-lived ไม่มีวันหมดอายุ แต่ถ้า Revoke หรือออก Token ใหม่ Token เก่าจะใช้งานไม่ได้ทันที ต้องอัปเดต Token ในระบบและแท็บ "ตั้งค่า"
